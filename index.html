        // 開啟測驗

        function startQuiz() {

            switchPage('welcomePage', 'quizPage');

        // 顯示題目 (核心功能：每次隨機凌亂分布選項)

        function showQuestion() {

            const q = questions[currentQuestionIndex];

            

            // 更新進度條

            const percent = ((currentQuestionIndex) / questions.length) * 100;

            document.getElementById('progressBar').style.width = `${percent}%`;


            // 置入題目文字

            document.getElementById('questionText').innerText = `Q${currentQuestionIndex + 1}. ${q.text}`;

            

            // 隨機打亂演算法 (Fisher-Yates Shuffle)

            const shuffledOptions = [...q.options];

            for (let i = shuffledOptions.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [shuffledOptions[i], shuffledOptions[j]] = [shuffledOptions[j], shuffledOptions[i]];

            }


            // 清空並渲染隨機排序後的卡片

            const optionsList = document.getElementById('optionsList');

            optionsList.innerHTML = '';

            

            shuffledOptions.forEach(opt => {

                const card = document.createElement('div');

                card.className = 'option-card';

                card.innerText = opt.text;

                card.onclick = () => selectOption(opt.type);

                optionsList.appendChild(card);

            });

        }


        // 選取選項邏輯

        function selectOption(type) {

            scores[type]++;

            currentQuestionIndex++;


            if (currentQuestionIndex < questions.length) {

                // 流暢淡入淡出切換下一題

                const quizPage = document.getElementById('quizPage');

                quizPage.style.opacity = '0';

                quizPage.style.transform = 'translateY(-5px)';

                

                setTimeout(() => {

                    showQuestion();

                    quizPage.style.opacity = '1';

                    quizPage.style.transform = 'translateY(0)';

                }, 250);

            } else {

                endQuiz();

            }

        }

            }

        // 換頁過渡輔助函式

        function switchPage(fromId, toId) {

            const fromPage = document.getElementById(fromId);

            const toPage = document.getElementById(toId);

            

            fromPage.classList.remove('active');

            setTimeout(() => {

                toPage.classList.add('active');

            }, 50);

        }
